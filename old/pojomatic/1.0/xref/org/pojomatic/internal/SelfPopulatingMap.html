6   /**
7    * A thread-safe "map" which generates values on demand, with the guarantee that no more than one
8    * value will be auto-created for a given key.
9    * Classes extending this class should override {@link #create(Object)}.
10   */
11  public abstract class SelfPopulatingMap<K, V> {
12  
13    public V get(final K key) {
14      V value = valueMap.get(key);
15      if (value == null) {
16        final Object mutex = new Object();
17        synchronized (mutex) {
18          Object existingMutex = mutexMap.putIfAbsent(key, mutex);
19          if (existingMutex == null) {
20            return tryCreatingValue(key);
21          }
22          else {
23            synchronized (existingMutex) {
24              V oldValue = valueMap.get(key);
25              // if the previous holder of this mutex failed to create a value, we'll give it a shot.
26              return oldValue != null ? oldValue : tryCreatingValue(key);
27            }
28          }
29        }
30      }
31      return value;
32    }
33  
34    /**
35     * Create a value for a key.  This will be called by {@link #get(Object)} when there is not
36     * already an existing value, and no other thread is already creating a value for that key.
37     * The value returned must not be null.
38     * @param key the key to create the value for
39     * @return the value
40     */
41    protected abstract V create(K key);
42  
43    private V tryCreatingValue(K key) {
44      V value = create(key);
45      valueMap.put(key, value);
46      return value;
47    }
48  
49    /**
50     * The values held by this map.
51     */
52    private final ConcurrentMap<K, V> valueMap = new ConcurrentHashMap<K, V>();
53  
54    /**
55     * Mutexes created on demand to ensure that only a single value is created for each key.
56     */
57    private final ConcurrentMap<K, Object> mutexMap = new ConcurrentHashMap<K, Object>();
58  }
