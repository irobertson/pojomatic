7    * The default property formatter used by Pojomatic.  While the particulars of the formatting
8    * strategy are subject to change, the general principle is to provide a meaningful representation.
9    * In particular, arrays are formatted "deeply", rather than simply showing the default toString
10   * representation of Java arrays.
11   */
12  public class DefaultPropertyFormatter implements PropertyFormatter {

14      //Not applicable

18    public String format(Object value) {
19      if (value == null) {
20        return "null";
21      }
22      else if (value.getClass().isArray()) {
23        Class<?> componentClass = value.getClass().getComponentType();
24        if (componentClass.isPrimitive()) {
25          if (Boolean.TYPE == componentClass) {
26            return Arrays.toString((boolean[]) value);
27          }
28          if (Character.TYPE == componentClass) {
29            StringBuilder builder = new StringBuilder().append('[');
30            boolean seenOne = false;
31            for (char c: ((char[]) value)) {
32              if(seenOne) {
33                builder.append(", ");
34              }
35              else {
36                seenOne = true;
37              }
38              builder.append('\'');
39              if (Character.isISOControl(c)) {
40                builder.append("0x").append(Integer.toHexString(c));
41              }
42              else {
43                builder.append(c);
44              }
45              builder.append('\'');
46            }
47            return builder.append(']').toString();
48          }
49          if (Byte.TYPE == componentClass) {
50            return Arrays.toString((byte[]) value);
51          }
52          if (Short.TYPE == componentClass) {
53            return Arrays.toString((short[]) value);
54          }
55          if (Integer.TYPE == componentClass) {
56            return Arrays.toString((int[]) value);
57          }
58          if (Long.TYPE == componentClass) {
59            return Arrays.toString((long[]) value);
60          }
61          if (Float.TYPE == componentClass) {
62            return Arrays.toString((float[]) value);
63          }
64          if (Double.TYPE == componentClass) {
65            return Arrays.toString((double[]) value);
66          }
67          else {
68            throw new IllegalStateException("unexpected primitive array base type: " + componentClass);
69          }
70        }
71        else {
72          return Arrays.deepToString((Object[]) value);
73        }
74      }
75      else {
76        return value.toString();
77      }
78    }
