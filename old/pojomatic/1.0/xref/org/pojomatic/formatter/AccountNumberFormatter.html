7   /**
8    * A property formatter which shows only the last 4 characters of the string representation of the
9    * {@link Property}, with all others replaced by an asterisk ('*').
10   * Useful for credit card numbers, social security numbers, etc.
11   * <p>
12   * For example, a 16 character {@code String} representing a credit card number would
13   * be formatted as "************1234".
14   */
15  public class AccountNumberFormatter extends DefaultPropertyFormatter {
16    private static final int DEFAULT_PLAINTEXT_CHARS = 4;
17    private static final int DEFAULT_FILL_CHAR = '*';
18  
19    private int plaintextChars = DEFAULT_PLAINTEXT_CHARS;
20    private char fillChar = DEFAULT_FILL_CHAR;
21  
22    @Override
23    public String format(Object value) {
24      String rep = super.format(value);
25      int repLength = rep.length();
26      if (repLength <= getPlaintextChars()) {
27        return rep;
28      } else {
29        char[] repChars = rep.toCharArray();
30        Arrays.fill(repChars, 0, repLength - getPlaintextChars(), getFillChar());
31        return String.valueOf(repChars);
32      }
33    }
34  
35    protected int getPlaintextChars() {
36      return plaintextChars;
37    }
38  
39    protected void setPlaintextChars(int plaintextChars) {
40      this.plaintextChars = plaintextChars;
41    }
42  
43    protected char getFillChar() {
44      return fillChar;
45    }

48      this.fillChar = fillChar;
49    }
