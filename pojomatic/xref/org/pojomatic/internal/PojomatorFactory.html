19    private static final class DynamicClassLoader extends ClassLoader {
20      private DynamicClassLoader(ClassLoader parent) {
21        super(parent);
22      }
23  
24      Class<?> loadClass(String name, byte[] classBytes) {
25        return defineClass(name, classBytes, 0, classBytes.length);
26      }
27    }
28  
29    private static volatile DynamicClassLoader CLASS_LOADER = null;
30    private final static Object CLASS_LOADER_MUTEX = new Object();
31  
32    // We don't initialize CLASS_LOADER at classLoad time because permission issues could cause an exception; we don't
33    // want that exception to be masked with a ClassNotFoundException
34    private static DynamicClassLoader getClassLoader() {
35      if (CLASS_LOADER == null) {
36        synchronized (CLASS_LOADER_MUTEX) {
37          if (CLASS_LOADER == null) {
38            CLASS_LOADER =
39              AccessController.doPrivileged(new PrivilegedAction<DynamicClassLoader>() {
40                @Override
41                public DynamicClassLoader run() {
42                  return new DynamicClassLoader(PojomatorFactory.class.getClassLoader());
43                }
44              });
45          }
46        }
47      }
48      return CLASS_LOADER;
49    }
50  
51    public static <T> Pojomator<T> makePojomator(final Class<T> pojoClass) {
52      try {
53        return AccessController.doPrivileged(new PrivilegedExceptionAction<Pojomator<T>>() {
54          @Override
55          public Pojomator<T> run() throws Exception {
56            return makePojomatorChecked(pojoClass);
57          }
58        });
59      } catch (PrivilegedActionException e) {
60        throw new RuntimeException(e.getCause());
61      }
62    }
63  
64    private static <T> Pojomator<T> makePojomatorChecked(Class<T> pojoClass)
65        throws IllegalAccessException, NoSuchFieldException, SecurityException, InstantiationException,
66        InvocationTargetException, NoSuchMethodException {
67      ClassProperties classProperties = ClassProperties.forClass(pojoClass);
68      PojomatorByteCodeGenerator generator = new PojomatorByteCodeGenerator(pojoClass, classProperties);
69      Class<?> pojomatorClass = getClassLoader().loadClass(generator.pojomatorClassName, generator.makeClassBytes());
70      @SuppressWarnings("unchecked")
71      Pojomator<T> pojomator = (Pojomator<T>) pojomatorClass.getConstructor(Class.class, ClassProperties.class)
72        .newInstance(pojoClass, classProperties);
73      for (PropertyElement propertyElement: classProperties.getToStringProperties()) {
74        setStaticField(
75          pojomatorClass,
76          PojomatorByteCodeGenerator.propertyFormatterName(propertyElement),
77          createPropertyFormatter(propertyElement.getElement()));
78      }
79      for (PropertyElement propertyElement: classProperties.getAllProperties()) {
80        setStaticField(pojomatorClass, PojomatorByteCodeGenerator.propertyElementName(propertyElement), propertyElement);
81      }
82      return pojomator;
83    }
84  
85    private static void setStaticField(Class<?> clazz, String fieldName, Object value)
86        throws NoSuchFieldException, SecurityException, IllegalAccessException {
87      Field field = clazz.getDeclaredField(fieldName);
88      field.setAccessible(true);
89      field.set(null, value);
90    }
91  
92    private static EnhancedPropertyFormatter createPropertyFormatter(AnnotatedElement annotatedElement)
93      throws InstantiationException, IllegalAccessException {
94      PropertyFormat propertyFormat = annotatedElement.getAnnotation(PropertyFormat.class);
95      EnhancedPropertyFormatter propertyFormatter = constructPropertyFormatter(propertyFormat);
96      propertyFormatter.initialize(annotatedElement);
97      return propertyFormatter;
98  
99    }
100 
101   private static EnhancedPropertyFormatter constructPropertyFormatter(PropertyFormat propertyFormat)
102     throws InstantiationException, IllegalAccessException {
103     if (propertyFormat == null) {
104       return new DefaultEnhancedPropertyFormatter();
105     }
106     else {
107       if (EnhancedPropertyFormatter.class.isAssignableFrom(propertyFormat.value())) {
108         return (EnhancedPropertyFormatter) propertyFormat.value().newInstance();
109       }
110       else {
111         @SuppressWarnings("deprecation")
112         EnhancedPropertyFormatterWrapper wrapper = new EnhancedPropertyFormatterWrapper(propertyFormat.value().newInstance());
113         return wrapper;
114       }
115     }
116   }
