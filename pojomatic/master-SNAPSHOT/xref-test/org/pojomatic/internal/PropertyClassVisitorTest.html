36    private static enum NameExtractor implements Function<PropertyElement, String> {
37      INSTANCE;
38      @Override
39      public String apply(PropertyElement p) {
40        AnnotatedElement element = p.getElement();
41        if (element instanceof Method) {
42          return ((Method) element).getName();
43        } else {
44          return ((Field) element).getName();
45        }
46      }
47    }

49    private PropertyElement f1;
50    private PropertyElement f2;
51    private PropertyElement f3;
52    private PropertyElement m1;
53    private PropertyElement m2;
54    private PropertyElement m3;
55    private PropertyElement m4;

57    private static List<PropertyElement> NO_PROPERTIES = Collections.emptyList();

59    @BeforeMethod
60    public void setup() throws Exception {
61      // Note that fields will be visited before methods, regardless of declaration order.
62      f1 = new PropertyField(FieldsAndGetters.class.getDeclaredField("field1"), "");
63      f2 = new PropertyField(FieldsAndGetters.class.getDeclaredField("field2"), "");
64      f3 = new PropertyField(Other.class.getDeclaredField("n"), "");
65      m1 = new PropertyAccessor(FieldsAndGetters.class.getDeclaredMethod("getter1"), "");
66      m2 = new PropertyAccessor(FieldsAndGetters.class.getDeclaredMethod("is2"), "");
67      m3 = new PropertyAccessor(FieldsAndGetters.class.getDeclaredMethod("m3", int.class), "");
68      m4 = new PropertyAccessor(FieldsAndGetters.class.getDeclaredMethod("m4", int.class), "");
69    }
70  
71    @Test
72    public void testReflectionOrdering() {
73      PropertyClassVisitor visitor = PropertyClassVisitor.visitClass(
74          FieldsAndGetters.class,
75          makeRoleMaps(Arrays.asList(f1, f2), Arrays.asList(f2, f1), NO_PROPERTIES),
76          makeRoleMaps(Arrays.asList(m2, m1), Arrays.asList(m1), Arrays.asList(m2)));
77      assertNotNull(visitor);
78      assertEquals(visitor.getSortedProperties(), makeRoleLists(Arrays.asList(f1, f2, m1, m2), Arrays.asList(f1, f2, m1), Arrays.asList(m2)));
79    }
80  
81    @Test
82    public void testMissingCodeSource() {
83      Map<PropertyRole, Map<String, PropertyElement>> roleMaps =
84          makeRoleMaps(NO_PROPERTIES, NO_PROPERTIES, NO_PROPERTIES);
85      PropertyClassVisitor visitor = PropertyClassVisitor.visitClass(String.class, roleMaps, roleMaps);
86      assertNull(visitor);
87    }

89    @Test
90    public void testMissingClassBytes() throws Exception {
91      class Bean {}
92  
93      ClassOnlyClassLoader classLoader = new ClassOnlyClassLoader(Bean.class.getClassLoader());
94      Class<?> beanClass = classLoader.loadClass(Bean.class.getName());
95      assertNotSame(Bean.class, beanClass);
96      assertNotNull(beanClass.getProtectionDomain().getCodeSource().getLocation());
97      Map<PropertyRole, Map<String, PropertyElement>> roleMaps =
98          makeRoleMaps(NO_PROPERTIES, NO_PROPERTIES, NO_PROPERTIES);
99      PropertyClassVisitor visitor = PropertyClassVisitor.visitClass(String.class, roleMaps, roleMaps);
100     assertNull(visitor);
101   }

103   @Test
104   public void testThrowReflectionMissmatch() throws Exception {
105     try {
106 
107       PropertyClassVisitor.visitClass(
108         FieldsAndGetters.class,
109         makeRoleMaps(Arrays.asList(f3), NO_PROPERTIES, NO_PROPERTIES),
110         makeRoleMaps(Arrays.asList(m3, m4), NO_PROPERTIES, NO_PROPERTIES));
111       fail("Exception expected");
112     }
113     catch (IllegalStateException e) {
114       assertEquals(e.getMessage(), "In class " + FieldsAndGetters.class.getName() + ", properties "
115       + f3.getElement().toString() + ", " + m3.getElement().toString() + ", " + m4.getElement().toString()
116       + " were found in reflection, but not when visiting the bytecode");
117     }
118   }

120   private static Map<PropertyRole, Map<String, PropertyElement>> makeRoleMaps(
121       List<PropertyElement> forEquals, List<PropertyElement> forHashCode, List<PropertyElement> forToString) {
122     EnumMap<PropertyRole, Map<String, PropertyElement>> enumMap = new EnumMap<>(PropertyRole.class);
123     enumMap.put(PropertyRole.EQUALS, Maps.uniqueIndex(forEquals, NameExtractor.INSTANCE));
124     enumMap.put(PropertyRole.HASH_CODE, Maps.uniqueIndex(forHashCode, NameExtractor.INSTANCE));
125     enumMap.put(PropertyRole.TO_STRING, Maps.uniqueIndex(forToString, NameExtractor.INSTANCE));
126     return enumMap;
127   }
128 
129   private static Map<PropertyRole, List<PropertyElement>> makeRoleLists(
130       List<PropertyElement> forEquals, List<PropertyElement> forHashCode, List<PropertyElement> forToString) {
131     EnumMap<PropertyRole, List<PropertyElement>> enumMap = new EnumMap<>(PropertyRole.class);
132     enumMap.put(PropertyRole.EQUALS, forEquals);
133     enumMap.put(PropertyRole.HASH_CODE, forHashCode);
134     enumMap.put(PropertyRole.TO_STRING, forToString);
135     return enumMap;
136   }
